### 思路
分区：从数组中选取一个值作为轴，把比它小的值放在左边，比它大的值放在右边。 
对轴两边的两个分区递归再进行分区。
分区的长度为0或1时，无需进行任何操作。 

### 以最右边的元素为轴
```
function quickSort(array, left, right) {
  //数组长度为1，返回该数组
  if (right - left < 1) {
    return array;
  } //以最右边的元素为轴
  let pivot = array[right]; //设定左右指针
  let L = left;
  let R = right - 1; //循环直到左右指针重合
  while (L <= R) {
    //移动左指针，如果左指针大于轴，停止
    while (array[L] <= pivot && L <= R) {
      L++;
    } //移动右指针，如果右指针小于轴，停止
    while (array[R] > pivot && L <= R) {
      R--;
    } //交换左右指针里的值
    if (L < R) {
      let temp = array[L];
      array[L] = array[R];
      array[R] = temp;
    }
  } //左右指针重合时，判断指针与轴的大小
  if (array[L] > pivot) {
    array[right] = array[L];
    array[L] = pivot;
  } //对左边的分区递归
  if (L - 1 - left >= 1) {
    quickSort(array, left, L - 1);
  } //对右边的分区递归
  if (right - L - 1 >= 1) {
    quickSort(array, L + 1, right);
  }
  return array;
}
```
